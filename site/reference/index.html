class SCI:
    """ SCI - Software Configuration Index.
    The overwhelming majority of this class was written by the 2021-2023 cohort.
    The 2023-2024 cohort made minor adjustments due to compatibility with OSes other than Windows.
    This documentation was written by the 2023-2024 based on the interpretation of what the code does, 
    but there may be some gaps in understanding.
    Some Docx methods are in src/helper/docxHelper.py, it is not clear why some methods are there and some are here.

    Additionally, IBM Doors Next does not seem to be compatible with Docx, so this class may require an overhaul
    """
    def __init__(self, config: InputDataPackage, artifacts: list, data_package_stages_config: list) -> None:
        """ Initialized by src/app/dataPackage.py
        Not clear what the logical separation between init and generate is.

        Args:
            config (InputDataPackage): Contains package name, output folder, and other properties not used. Output folder and package name used to generate file
            artifacts (list): artifacts (files) in BitBucket, collected in DataPackage
            data_package_stages_config (list): Only Stages are used.
        """        
        self.config: InputDataPackage = config
        self.stages_config: list = data_package_stages_config

        self.artifacts: list = artifacts

        self.artifacts_specification_stage = []
        self.artifacts_sw_implementation_stage = []
        self.artifacts_sw_testing_stage = []
        self.artifacts_hsi_system_testing_stage = []
        self.artifacts_miscellaneous = []

        self.warnings = []
        self.errors = []

        stages = []
        for stage in self.stages_config['stages']:
            stages.append(stage['title'])

        for artifact in self.artifacts:
            file_path = artifact['file_path_data_package']

            # The stage is always the second before last
            artifact_stage = file_path.split(os.sep)[-2]

            # The stage index are in order of the data_package_stages.json file
            # Spec-Stage
            if stages[0] == artifact_stage:
                self.artifacts_specification_stage.append(artifact)
            # SW-Stage
            elif stages[1] == artifact_stage:
                self.artifacts_sw_implementation_stage.append(artifact)
            # SW-Test-Stage
            elif stages[2] == artifact_stage:
                self.artifacts_sw_testing_stage.append(artifact)
            # HSI-and-Sys-Test-Stage
            elif stages[3] == artifact_stage:
                self.artifacts_hsi_system_testing_stage.append(artifact)
            # Miscellaneous
            else:
                self.artifacts_miscellaneous.append(artifact)
                self.warnings.append(
                    "- The artifact [" + artifact['file_name'] + "] was not associated to a Stage")

    def generate(self) -> None:
        """ Actually generates the SCI.
        This uses a number of docx methods, which are not very readable.
        This method should be refactored int
        """        

        document = Document()               #create empty docx document

        sections = document.sections
        # Template Margins
        for section in sections:
            section.top_margin = Cm(1.75)
            section.bottom_margin = Cm(3)
            section.left_margin = Cm(1.5)
            section.right_margin = Cm(1.5)

        # Set Doc Header Title
        paragraph = document.add_paragraph()
        paragraph.paragraph_format.alignment = WD_PARAGRAPH_ALIGNMENT.CENTER
        run = paragraph.add_run('Software Configuration Index Report')
        run.font.name = 'Calibri'
        run.font.size = Pt(16)
        run.font.bold = True
        run.font.underline = True

        document.add_paragraph('')

        self.check_empty_stages()

        # TODO: add try catch exception
        SCI.add_section_reference_table(self, document)
        SCI.add_section_sci_warnings_errors(
            list=self.errors, heading_text="SCI Errors", document=document)
        SCI.add_section_sci_warnings_errors(
            list=self.warnings, heading_text="SCI Warnings", document=document)

        document.add_heading('SCI Data', level=2)
        artifacts = []
        for index, stage_config in enumerate(self.stages_config['stages']):
            match index:
                # Software Implementation Stage
                case 0:
                    artifacts = self.artifacts_specification_stage
                # Software Implementation Stage
                case 1:
                    artifacts = self.artifacts_sw_implementation_stage
                # Software Testing Stage
                case 2:
                    artifacts = self.artifacts_sw_testing_stage
                # HSI & System Testing Stage
                case 3:
                    artifacts = self.artifacts_hsi_system_testing_stage
            SCI.add_subsection_artifacts_stage(self, config_index=index,
                                               artifacts=artifacts,
                                               document=document,
                                               is_misc=False
                                               )
            document.add_paragraph()

       # Miscellaneous
        SCI.add_subsection_artifacts_stage(self, config_index=-1,
                                           artifacts=self.artifacts_miscellaneous,
                                           document=document,
                                           is_misc=True
                                           )

        DocxHelper.add_page_number(
            document.sections[0].footer.paragraphs[0].add_run())
        document.sections[0].footer.paragraphs[0].alignment = WD_PARAGRAPH_ALIGNMENT.RIGHT

        sci_path = os.path.join(self.config.output_folder, self.config.package_name, '{}_SCI_Report.docx'
                                .format(self.config.package_name))
        # Saving Docx
        try:
            document.save(sci_path)
        except PermissionError:
            Logger.error(
                f'Permission Error during saving of SCI File. SCI was not updated [{sci_path}]')

            raise Exception()
        except:     # to catch Exceptions that are not PermissionError
            raise Exception()

    def check_empty_stages(self) -> None:
        """Adds warnings about stages without any artifacts
        """        
        if not self.artifacts_specification_stage:
            self.warnings.append(
                CommonValues.WARNING_NO_ARTIFACTS_IN_STAGE % CommonValues.SPECIFICATION_STAGE_NAME)
        if not self.artifacts_sw_implementation_stage:
            self.warnings.append(CommonValues.WARNING_NO_ARTIFACTS_IN_STAGE %
                                 CommonValues.SW_IMPLEMENTATION_STAGE_NAME)
        if not self.artifacts_sw_testing_stage:
            self.warnings.append(
                CommonValues.WARNING_NO_ARTIFACTS_IN_STAGE % CommonValues.SW_TESTING_STAGE_NAME)
        if not self.artifacts_hsi_system_testing_stage:
            self.warnings.append(CommonValues.WARNING_NO_ARTIFACTS_IN_STAGE %
                                 CommonValues.HSI_SYSTEM_TESTING_STAGE_NAME)

    def create_empty_file(self, stage: str) -> None:
        """Generates a file empty.txt inside the folder of the corresponding stage.
        Used to indicate that there are no artifacts associated to a stage.
        Likely an arbitrary decision to ensure that the code ran and actively tried to find artifacts, but did not.
        An actual empty folder might have been taken for an error, which maybe explains the decision process.

        Args:
            stage (str): Name of the stage
        """        
        stage_path = os.path.abspath(os.path.join(
            self.config.output_folder, self.config.package_name, stage))
        file_path = os.path.join(stage_path, EMPTY_FILE_NAME_AND_TYPE)

        if not os.path.isdir(stage_path):
            FileHelper.create_dir(stage_path)

        if os.path.isfile(file_path):
            FileHelper.remove_file(file_path)

        empty_file = open(file_path, mode="w+")
        empty_file.close()
        Logger.info(f'Created file [{file_path}]')

        stage_path = os.path.join(output_path, stage)
        if not os.path.isdir(stage_path):
            FileHelper.create_dir(stage_path)

        if os.path.isfile(file_path):

    def add_section_reference_table(self, document: Document) -> None:
        """Adds a reference table at the top of a document

        Args:
            document (docx.Document): document where to add reference table
        """        
        # SCI - Reference Table
        sci_ref_table = document.add_table(rows=4, cols=2)
        sci_ref_table.style = 'Table Grid'
        section_table = {
            'Reference ID:': self.config.package_name,
            'Generation Date:': datetime.now().strftime('%d/%m/%Y %H:%M:%S'),
            '#nb of errors:': len(self.errors),
            '#nb of Warnings:': len(self.warnings)
        }
        for i, (label, content) in enumerate(section_table.items()):
            table_cells = sci_ref_table.rows[i].cells
            table_cells[0].width = Cm(4.5)
            DocxHelper.set_paragraph_as_table_header(table_cells[0].paragraphs[0],
                                                     label,
                                                     TABLE_HEADER_FONT_SIZE,
                                                     WD_PARAGRAPH_ALIGNMENT.LEFT)
            table_cells[1].text = str(content)

        document.add_paragraph()

    @staticmethod
    def add_subsection(document: Document, title: str, rows: int = 2) :
        """ Adds Table.
        Not sure why method is called add_subsection.
        Each subsection has one table, which might explain the overlapping nomenclature.

        Args:
            document (Document): document where to add the table
            title (str): Title of the table
            rows (int, optional): number of rows to add. Defaults to 2.

        Returns:
            Document.addtable: Returns a table
        """        
        sci_data_table = document.add_table(rows=rows, cols=6)
        sci_data_table.autofit = True
        sci_data_table.style = 'Table Grid'

        table_cells = sci_data_table.rows[0].cells
        first_col = table_cells[0]
        last_col = table_cells[5]
        first_col.merge(last_col)
        # Adds main title of the table
        DocxHelper.set_paragraph_as_table_header(table_cells[0].paragraphs[0],
                                                 title,
                                                 TABLE_TITLE_FONT_SIZE)

        # Adds subheading on row immediately below main title
        table_cells = sci_data_table.rows[1].cells
        for i, h in enumerate(SUBSECTION_COLS):
            DocxHelper.set_paragraph_as_table_header(table_cells[i].paragraphs[0],
                                                     h,
                                                     TABLE_HEADER_FONT_SIZE)

        return sci_data_table

    def add_subsection_artifacts_stage(self, document: Document, config_index: Literal[-1,0,1,2,3], artifacts: list, is_misc: bool) -> None:
        """Adds tables for artifact stages. If a stage is empty, requests creation of empty file.
        First it requests a table to be created with the title and headers, then adds rows with the data

        Args:
            document (Document): document where to insert tables
            config_index (int): index of the config. -1 was used in the Miscellaneous call, but not used. 0~3 refer to the stages
            artifacts (list): list of artifacts of the stage
            is_misc (bool): True: miscellaneous. False: Associated to a stage
        """        
        rows = 2 + len(artifacts)

        table_header_text = ''
        folder = ''
        if is_misc:
            misc = self.stages_config['miscellaneous']
            folder = misc['title']
            table_header_text = misc['sci_table_title']
        else:
            stages = self.stages_config['stages']
            folder = stages[config_index]['title']
            table_header_text = stages[config_index]['sci_table_title']

        sci_data_table = SCI.add_subsection(
            document=document, title=table_header_text, rows=rows)

        if not artifacts:
            self.create_empty_file(stage=folder)
            return

        # 'Filename', 'VCS Commit / Version', 'VCS Location', 'Jira ID', 'Lifecycle', 'Responsible' - 07/28/2022
        for i, artifact in enumerate(artifacts):
            self.create_file(artifact)
            table_cells = sci_data_table.rows[2 + i].cells
            for j, header in enumerate(SUBSECTION_COLS):
                # Filename
                if header == SUBSECTION_COLS[0]:
                    table_cells[j].text = artifact['file_name']

                # VCS Commit / Version
                elif header == SUBSECTION_COLS[1]:
                    p_table = table_cells[j].add_paragraph()
                    DocxHelper.add_hyperlink(
                        p_table, artifact['last_commit_url'], artifact['last_commit_id'])

                # VCS Location
                elif header == SUBSECTION_COLS[2]:
                    p_table = table_cells[j].add_paragraph()
                    DocxHelper.add_hyperlink(
                        p_table, artifact['file_path_repo_url'], artifact['file_path_repo'])

                # Jira ID
                elif header == SUBSECTION_COLS[3]:
                    p_table = table_cells[j].add_paragraph()
                    DocxHelper.add_hyperlink(
                        p_table, artifact['ticket_url'], artifact['ticket_key'])

                # Lifecycle
                elif header == SUBSECTION_COLS[4]:
                    if is_misc:
                        table_cells[j].text = "N/A"
                    else:
                        filename = artifact['file_path_data_package'].split(
                            "_")[-1]
                        lifecycle = filename.split(".")[0]
                        table_cells[j].text = lifecycle

                # Responsible
                elif header == SUBSECTION_COLS[5]:
                    if artifact['assignee_profile_url'] == 'N/A':
                        table_cells[j].text = "N/A"
                    else:
                        p_table = table_cells[j].add_paragraph()
                        DocxHelper.add_hyperlink(
                            p_table, artifact['assignee_profile_url'], artifact['assignee_name'])

        document.add_paragraph()

    @staticmethod
    def add_section_sci_warnings_errors(list: list[str], heading_text: str, document: Document) -> None:
        """Creates table with SCI warnings or errors

        Args:
            list (list[str]): list of errors
            heading_text (str): Title of the Table (will be placed before the table, OUTSIDE the table, as a text)
            document (Document): document where to add the title and the table
        """        
        # SCI Warnings
        document.add_heading(heading_text, level=2)

        table_row_number = len(list) if list else 1
        sci_table = document.add_table(rows=table_row_number, cols=1)

        sci_table.style = 'Table Grid'
        table_cells = sci_table.rows[0].cells
        if not list:
            DocxHelper.set_cell_border_default(table_cells[0])
            return

        for i, sentence in enumerate(list):
            table_cells = sci_table.rows[i].cells
            DocxHelper.set_cell_border_default(table_cells[0])
            table_cells[0].text = sentence

        config_index (int): index of the config. -1 was used in the Miscellaneous call, but not used. 0~3 refer to the stages
        artifacts (list): list of artifacts of the stage
        is_misc (bool): True: miscellaneous. False: Associated to a stage
    """        
    rows = 2 + len(artifacts)

    table_header_text = ''
    folder = ''
    if is_misc:
        misc = self.stages_config['miscellaneous']
        folder = misc['title']
        table_header_text = misc['sci_table_title']
    else:
        stages = self.stages_config['stages']
        folder = stages[config_index]['title']
        table_header_text = stages[config_index]['sci_table_title']

    sci_data_table = SCI.add_subsection(
        document=document, title=table_header_text, rows=rows)

    if not artifacts:
        self.create_empty_file(stage=folder)
        return

    # 'Filename', 'VCS Commit / Version', 'VCS Location', 'Jira ID', 'Lifecycle', 'Responsible' - 07/28/2022
    for i, artifact in enumerate(artifacts):
        self.create_file(artifact)
        table_cells = sci_data_table.rows[2 + i].cells
        for j, header in enumerate(SUBSECTION_COLS):
            # Filename
            if header == SUBSECTION_COLS[0]:
                table_cells[j].text = artifact['file_name']

            # VCS Commit / Version
            elif header == SUBSECTION_COLS[1]:
                p_table = table_cells[j].add_paragraph()
                DocxHelper.add_hyperlink(
                    p_table, artifact['last_commit_url'], artifact['last_commit_id'])

            # VCS Location
            elif header == SUBSECTION_COLS[2]:
                p_table = table_cells[j].add_paragraph()
                DocxHelper.add_hyperlink(
                    p_table, artifact['file_path_repo_url'], artifact['file_path_repo'])

            # Jira ID
            elif header == SUBSECTION_COLS[3]:
                p_table = table_cells[j].add_paragraph()
                DocxHelper.add_hyperlink(
                    p_table, artifact['ticket_url'], artifact['ticket_key'])

            # Lifecycle
            elif header == SUBSECTION_COLS[4]:
                if is_misc:
                    table_cells[j].text = "N/A"
                else:
                    filename = artifact['file_path_data_package'].split(
                        "_")[-1]
                    lifecycle = filename.split(".")[0]
                    table_cells[j].text = lifecycle

            # Responsible
            elif header == SUBSECTION_COLS[5]:
                if artifact['assignee_profile_url'] == 'N/A':

                else:
                    p_table = table_cells[j].add_paragraph()
                    DocxHelper.add_hyperlink(
                        p_table, artifact['assignee_profile_url'], artifact['assignee_name'])

    document.add_paragraph()

class InputFunctions:
    """Class that contains methods to help with input methods
    """    
    @staticmethod
    def is_valid_folder_name(folder_name: str, max_length: int = 255) -> bool:
        """Checks if a string is a valid Windows folder name according to the specifications here:
        https://learn.microsoft.com/en-us/windows/win32/fileio/naming-a-file

        Args:
            folder_name (str): name of the folder
            max_length (int, optional): Max length of the string in the OS. Defaults to 255.

        Returns:
            bool: True if all checks pass
        """        
        # Check length of folder name
        max_length = max_length
        if len(folder_name) > max_length:
            return False
        if len(folder_name) < 1:
            return False

        # Check for invalid characters
        if InputFunctions.check_reserved_characters(text = folder_name):
            return False

        # Check for strings being correctly passed:
        if not InputFunctions.check_is_str(text = folder_name):
            return False

        # Check if name ends with a period or space or is composed entirely of periods or spaces
        if InputFunctions.check_name_ending(text = folder_name):
            return False

        # All constraints met, folder name is valid
        return True

    def is_valid_file_name(file_name: str, max_length: int = 255) -> bool:
        """Checks if a string is a valid Windows file name according to the specifications here:
        https://learn.microsoft.com/en-us/windows/win32/fileio/naming-a-file

        Args:
            file_name (str): name of the file
            max_length (int, optional): Max length of the string in the OS. Defaults to 255.

        Returns:
            bool: True if all checks pass
        """         
        # Check length of file name
        max_length = max_length
        if len(file_name) > max_length:
            return False
        if len(file_name) < 1:
            return False

        if InputFunctions.check_reserved_characters(file_name):
            return False

        # Check if file name ends with a period or space or is composed entirely of periods or spaces
        if InputFunctions.check_name_ending(file_name):
            return False

        # Check if file name ends with any of the reserved device names
        reserved_names = {'CON', 'PRN', 'AUX', 'NUL'}
        for i in range(10):
            reserved_names.add(f'COM{i}')
            reserved_names.add(f'LPT{i}')
        for i in ['¹', '²', '³']:
            reserved_names.add(f'COM{i}')
            reserved_names.add(f'LPT{i}')
        for reserved_name in reserved_names:
            if file_name.upper().endswith(reserved_name.upper()):
                return False

        # All constraints met, file name is valid
        return True

    @staticmethod
    def check_reserved_characters(text: str, reserved_chars:list[str] = r'[<>.:"/\\|?*]') -> bool:
        """Checks if a string has any reserved characters.
        Created originally to check if a string contains any characters reserved in file or folder names.
        See more: https://learn.microsoft.com/en-us/windows/win32/fileio/naming-a-file

        Args:
            text (str): string to check against the reserved characters
            reserved_chars (_type_, optional): list of characters. Defaults to r'[<>.:"/\|?*]'.

        Returns:
            bool: True if none of the characters is present.
        """        
        # Check for reserved characters
        reserved_chars = reserved_chars

def is_valid_file_name(file_name: str, max_length: int = 255) -> bool:
    """Checks if a string is a valid Windows file name according to the specifications here:
    https://learn.microsoft.com/en-us/windows/win32/fileio/naming-a-file

    Args:
        file_name (str): name of the file
        max_length (int, optional): Max length of the string in the OS. Defaults to 255.

    Returns:
        bool: True if all checks pass
    """         
    # Check length of file name
    max_length = max_length
    if len(file_name) > max_length:
        return False
    if len(file_name) < 1:
        return False

    if InputFunctions.check_reserved_characters(file_name):
        return False

    # Check if file name ends with a period or space or is composed entirely of periods or spaces
    if InputFunctions.check_name_ending(file_name):
        return False

    # Check if file name ends with any of the reserved device names
    reserved_names = {'CON', 'PRN', 'AUX', 'NUL'}
    for i in range(10):
        reserved_names.add(f'COM{i}')
        reserved_names.add(f'LPT{i}')
    for i in ['¹', '²', '³']:
        reserved_names.add(f'COM{i}')
        reserved_names.add(f'LPT{i}')
    for reserved_name in reserved_names:
        if file_name.upper().endswith(reserved_name.upper()):
            return False

    # All constraints met, file name is valid
    return True

class JiraRequests:
    """Authenticates with Jira
    """    
    MAX_RETRIES = 3
    TIMEOUT = 30  # in seconds

    authentication_type: AuthType = AuthType.SESSION_COOKIE
    session_cookie = ""
    session = Session()

    @staticmethod
    def get(url: str) -> requests.Response:
        """Used by several jira/jira methods

        Args:
            url (str): endpoint to get

        Returns:
            requests.Response: response
        """        
        for retry in range(JiraRequests.MAX_RETRIES):
            response = JiraRequests.session_get(url)

            if response.status_code == HttpStatusHelper.OK.value:
                return response

            if JiraRequests.authentication_type == AuthType.SESSION_COOKIE and JiraRequests.__is_session_cookie_invalid(
                    response):
                JiraRequests.__refresh_session_cookie()

        # Throw error if the request fails after trying MAX_RETRIES
        JiraRequests.__raise_error(response)

    @staticmethod
    def session_get(url: str) -> requests.Response:
        """ Gets session in Jira before using get

        Args:
            url (str): url to request session

        Returns:
            requests.Response: response
        """        
        jira_config = Config.jira()
        auth = None

        if JiraRequests.authentication_type == AuthType.SESSION_COOKIE:
            session_cookie = JiraRequests.session_cookie
            JiraRequests.session.headers.update({'cookie': session_cookie})

        if JiraRequests.authentication_type == AuthType.BASIC_AUTH:
            auth = (jira_config['auth']['username'],
                    jira_config['auth']['token'])

        return JiraRequests.session.get(url=url, auth=auth, timeout=JiraRequests.TIMEOUT)

    @staticmethod
    def __is_session_cookie_invalid(request_response: requests.Response) -> bool:
        """Verifies if session cookie is invalid

        Args:
            request_response (requests.Response): response

        Returns:
            bool: True if invalid
        """        
        return request_response.status_code == HttpStatusHelper.UNAUTHORIZED.value or len(
            JiraRequests.session_cookie) == 0

    @staticmethod
    def __refresh_session_cookie() -> None:
        """Refreshes session cookie. Used if cookie is invalid
        """        
        authentication_response = JiraRequests.__session_cookie_authentication()
        authentication_session_dict = json.loads(
            authentication_response.text)['session']
        session_cookie = '{}={}'.format(
            authentication_session_dict['name'], authentication_session_dict['value'])
        JiraRequests.session_cookie = session_cookie
        FileHelper.save_data({
            'session_cookie': session_cookie
        })

    @staticmethod
    def __session_cookie_authentication() -> requests.Response:
        """Used to refresh session cookie.
        Uses username and password (not bearer token)

        Returns:
            requests.Response: response
        """        
        jira_config = Config.jira()
        payload = {
            'username': jira_config['auth']['username'],
            'password': jira_config['auth']['password']
        }
        headers = {
            'Content-Type': 'application/json'
        }
        Logger.info('Authenticating with Jira...')
        response = requests.post('https://{}/rest/auth/1/session'.format(jira_config['domain']), headers=headers,
                                 data=json.dumps(payload), timeout=JiraRequests.TIMEOUT)
        Logger.info('Authenticated with Jira...')

        if response.status_code != HttpStatusHelper.OK.value:
            JiraRequests.__raise_error(response)

        return response

    @staticmethod
    def load_session_cookie() -> None:
        """ Loads session cookie
        """        
        try:
            data = FileHelper.load_data()
            JiraRequests.session_cookie = data['session_cookie']
        except:
            JiraRequests.session_cookie = ''

    @staticmethod
    def __raise_error(response: requests.Response) -> None:
        """Logs errors in responses

        Args:
            response (requests.Response): response
        """        
        error_message = 'Unknown error details'
        try:
            errors = response.json()['errors']
            error_message = errors[0]['message']
        except:
            Logger.error('Error executing the requests [{}]'.format(
                response.request.url))

def __session_cookie_authentication() -> requests.Response:
    """Used to refresh session cookie.
    Uses username and password (not bearer token)

    Returns:
        requests.Response: response
    """        
    jira_config = Config.jira()
    payload = {
        'username': jira_config['auth']['username'],
        'password': jira_config['auth']['password']
    }
    headers = {
        'Content-Type': 'application/json'
    }
    Logger.info('Authenticating with Jira...')
    response = requests.post('https://{}/rest/auth/1/session'.format(jira_config['domain']), headers=headers,
                             data=json.dumps(payload), timeout=JiraRequests.TIMEOUT)
    Logger.info('Authenticated with Jira...')

    if response.status_code != HttpStatusHelper.OK.value:
        JiraRequests.__raise_error(response)

    return response

    for retry in range(JiraRequests.MAX_RETRIES):
        response = JiraRequests.session_get(url)

@staticmethod
def session_get(url: str) -> requests.Response:
    """ Gets session in Jira before using get

    Args:
        url (str): url to request session

    Returns:
        requests.Response: response
    """        
    jira_config = Config.jira()
    auth = None

    if JiraRequests.authentication_type == AuthType.SESSION_COOKIE:
        session_cookie = JiraRequests.session_cookie
        JiraRequests.session.headers.update({'cookie': session_cookie})

    if JiraRequests.authentication_type == AuthType.BASIC_AUTH:
        auth = (jira_config['auth']['username'],
                jira_config['auth']['token'])

    return JiraRequests.session.get(url=url, auth=auth, timeout=JiraRequests.TIMEOUT)
